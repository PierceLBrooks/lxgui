422 void text::update_lines_()
423 {
424  // Update the line list, read format tags, do word wrapping, ...
425  lLineList_.clear();
426  lFormatList_.clear();
427 
428  DEBUG_LOG(" Get max line nbr");
429  uint uiMaxLineNbr, uiCounter = 0;
430  if (fBoxH_ != 0.0f && !std::isinf(fBoxH_))
431  {
432  if (fBoxH_ < get_line_height())
433  {
434  uiMaxLineNbr = 0;
435  }
436  else
437  {
438  float fRemaining = fBoxH_ - get_line_height();
439  uiMaxLineNbr = 1 + floor(fRemaining/(get_line_height()*fLineSpacing_));
440  }
441  }
442  else
443  uiMaxLineNbr = uint(-1);
444 
445  if (uiMaxLineNbr != 0)
446  {
447  std::vector<utils::ustring> lManualLineList = utils::cut_each(sUnicodeText_, U"\n");
448  for (auto iterManual : utils::range::iterator(lManualLineList))
449  {
450  const utils::ustring& sManualLine = *iterManual;
451 
452  DEBUG_LOG(" Line : '" + utils::unicode_to_UTF8(sManualLine) + "'");
453  // Make a temporary line array
454  std::vector<line> lLines;
455  line mLine; mLine.fWidth = 0.0f;
456  std::map<uint, format> lTempFormatList;
457 
458  DEBUG_LOG(" Read chars");
459  for (auto iterChar1 = sManualLine.begin(); iterChar1 != sManualLine.end(); ++iterChar1)
460  {
461  DEBUG_LOG(" char '" + utils::to_string(*iterChar1) + "'");
462  DEBUG_LOG(" Read format");
463  // Read format tags
464  if (*iterChar1 == U'|' && bFormattingEnabled_)
465  {
466  ++iterChar1;
467  if (iterChar1 == sManualLine.end()) break;
468 
469  if (*iterChar1 != U'|')
470  {
471  text::format mFormat;
472  if (get_format(iterChar1, sManualLine.end(), mFormat))
473  lTempFormatList[uiCounter+mLine.sCaption.size()] = mFormat;
474 
475  if (iterChar1 != sManualLine.end())
476  ++iterChar1;
477  if (iterChar1 == sManualLine.end()) break;
478  }
479  }
480 
481  DEBUG_LOG(" Get width");
482  if (*iterChar1 == U' ')
483  mLine.fWidth += fSpaceWidth_ + fTracking_;
484  else if (*iterChar1 == U'\t')
485  mLine.fWidth += 4*fSpaceWidth_ + fTracking_;
486  else
487  {
488  mLine.fWidth += get_character_width(*iterChar1) + fTracking_;
489  auto iterNext = iterChar1 + 1;
490  if (iterNext != sManualLine.end())
491  {
492  if (*iterNext != U' ')
493  mLine.fWidth += get_character_kerning(*iterChar1, *iterNext);
494  }
495  }
496 
497  mLine.sCaption += *iterChar1;
498 
499  if (mLine.fWidth > fBoxW_)
500  {
501  DEBUG_LOG(" Box break " + utils::to_string(mLine.fWidth) + " > " + utils::to_string(fBoxW_));
502  // Whoops, the line is too long...
503  if (mLine.sCaption.find(U" ") != mLine.sCaption.npos && bWordWrap_)
504  {
505  DEBUG_LOG(" Spaced");
506  // There are several words on this line, we'll
507  // be able to put the last one on the next line
508  utils::ustring::iterator iterChar2 = mLine.sCaption.end();
509  utils::ustring sErasedString;
510  uint uiCharToErase = 0;
511  float fErasedWidth = 0.0f;
512  bool bLastWasWord = false;
513  while (mLine.fWidth > fBoxW_ && iterChar2 != mLine.sCaption.begin())
514  {
515  --iterChar2;
516  if (*iterChar2 == U' ')
517  {
518  if (!bLastWasWord || bRemoveStartingSpaces_ || mLine.fWidth - fErasedWidth > fBoxW_)
519  {
520  mLine.fWidth -= fErasedWidth + fSpaceWidth_ + fTracking_;
521  sErasedString.insert(sErasedString.begin(), *iterChar2);
522  fErasedWidth = 0.0f;
523  ++uiCharToErase;
524  }
525  else
526  break;
527  }
528  else
529  {
530  fErasedWidth += get_character_width(*iterChar2) + fTracking_;
531  sErasedString.insert(sErasedString.begin(), *iterChar2);
532  ++uiCharToErase;
533  bLastWasWord = true;
534  }
535  }
536 
537  if (bRemoveStartingSpaces_)
538  {
539  while (iterChar2 != mLine.sCaption.end() && *iterChar2 == U' ')
540  {
541  --uiCharToErase;
542  sErasedString.erase(0, 1);
543  ++iterChar2;
544  }
545  }
546 
547  mLine.sCaption.erase(mLine.sCaption.size() - uiCharToErase, uiCharToErase);
548 
549  lLines.push_back(mLine);
550  for (auto& mFormat : lTempFormatList)
551  lFormatList_.insert(std::move(mFormat));
552 
553  lTempFormatList.clear();
554  uiCounter += mLine.sCaption.size();
555  mLine.fWidth = get_string_width(sErasedString);
556  mLine.sCaption = sErasedString;
557  }
558  else
559  {
560  DEBUG_LOG(" Single word");
561  // There is only one word on this line, or word
562  // wrap is disabled. Anyway, this line is just
563  // too long for the text box : our only option
564  // is to truncate it.
565  if (bAddEllipsis_)
566  {
567  DEBUG_LOG(" Ellipsis");
568  float fWordWidth = 3.0f*(get_character_width(U'.') + fTracking_);
569  utils::ustring::iterator iterChar2 = mLine.sCaption.end();
570  uint uiCharToErase = 0;
571  while ((mLine.fWidth + fWordWidth > fBoxW_) && (iterChar2 != mLine.sCaption.begin()))
572  {
573  --iterChar2;
574  mLine.fWidth -= get_character_width(*iterChar2) + fTracking_;
575  ++uiCharToErase;
576  }
577 
578  DEBUG_LOG(" Char to erase : " + utils::to_string(uiCharToErase) + " / "
579  + utils::to_string(mLine.sCaption.size()));
580 
581  mLine.sCaption.erase(mLine.sCaption.size() - uiCharToErase, uiCharToErase);
582  mLine.sCaption += U"...";
583  }
584  else
585  {
586  DEBUG_LOG(" Truncate");
587  utils::ustring::iterator iterChar2 = mLine.sCaption.end();
588  uint uiCharToErase = 0;
589  while (mLine.fWidth > fBoxW_ && iterChar2 != mLine.sCaption.begin())
590  {
591  --iterChar2;
592  mLine.fWidth -= get_character_width(*iterChar2) + fTracking_;
593  ++uiCharToErase;
594  }
595  mLine.sCaption.erase(mLine.sCaption.size() - uiCharToErase, uiCharToErase);
596  }
597 
598  if (!bWordWrap_)
599  {
600  DEBUG_LOG(" Display single line");
601  // Word wrap is disabled, so we can only display one line anyway.
602  lLineList_.push_back(mLine);
603  for (const auto& mFormat : lTempFormatList)
604  lFormatList_[mFormat.first] = mFormat.second;
605 
606  return;
607  }
608 
609  DEBUG_LOG(" Continue");
610  auto iterTemp = iterChar1;
611  size_t pos = sManualLine.find(U" ", iterChar1 - sManualLine.begin());
612  if (pos == sManualLine.npos)
613  iterChar1 = sManualLine.end();
614  else
615  iterChar1 = sManualLine.begin() + pos;
616 
617  if (iterChar1 != sManualLine.end())
618  {
619  // Read cut format tags
620  if (bFormattingEnabled_)
621  {
622  while (iterTemp != iterChar1)
623  {
624  if (*iterTemp == U'|')
625  {
626  ++iterTemp;
627  if (iterTemp == iterChar1) break;
628 
629  if (*iterTemp != U'|')
630  {
631  text::format mFormat;
632  if (get_format(iterTemp, iterChar1, mFormat))
633  {
634  lTempFormatList[uiCounter+mLine.sCaption.size()] = mFormat;
635  }
636  }
637  }
638 
639  if (iterTemp != iterChar1)
640  ++iterTemp;
641  }
642  }
643 
644  // Look for the next word
645  while (iterChar1 != sManualLine.end())
646  {
647  if ((*iterChar1) == U' ')
648  ++iterChar1;
649  else
650  break;
651  }
652 
653  // Add the line
654  if (iterChar1 != sManualLine.end())
655  {
656  --iterChar1;
657  lLines.push_back(mLine);
658  uiCounter += mLine.sCaption.size();
659 
660  for (auto& mFormat : lTempFormatList)
661  lFormatList_.insert(std::move(mFormat));
662 
663  lTempFormatList.clear();
664  mLine.fWidth = 0.0f;
665  mLine.sCaption.clear();
666  }
667  else
668  break;
669  }
670  else
671  break;
672  }
673  }
674  }
675 
676  DEBUG_LOG(" End");
677 
678  if (iterManual != lManualLineList.end() - 1)
679  mLine.sCaption += U"\n";
680 
681  lLines.push_back(mLine);
682  uiCounter += mLine.sCaption.size();
683 
684  for (auto& mFormat : lTempFormatList)
685  lFormatList_.insert(std::move(mFormat));
686 
687  lTempFormatList.clear();
688 
689  // Add the maximum number of line to this text
690  for (auto& sLine : lLines)
691  {
692  lLineList_.push_back(std::move(sLine));
693  if (lLineList_.size() == uiMaxLineNbr)
694  return;
695  }
696  DEBUG_LOG(" .");
697  }
