425 void text::update_lines_()
426 {
427  // Update the line list, read format tags, do word wrapping, ...
428  lLineList_.clear();
429  lFormatList_.clear();
430 
431  DEBUG_LOG(" Get max line nbr");
432  uint uiMaxLineNbr, uiCounter = 0;
433  if (fBoxH_ != 0.0f && !math::isinf(fBoxH_))
434  {
435  if (fBoxH_ < get_line_height())
436  {
437  uiMaxLineNbr = 0;
438  }
439  else
440  {
441  float fRemaining = fBoxH_ - get_line_height();
442  uiMaxLineNbr = 1 + floor(fRemaining/(get_line_height()*fLineSpacing_));
443  }
444  }
445  else
446  uiMaxLineNbr = uint(-1);
447 
448  if (uiMaxLineNbr != 0)
449  {
450  std::vector<utils::ustring> lManualLineList = utils::cut_each(sUnicodeText_, U"\n");
451  for (auto iterManual : utils::range::iterator(lManualLineList))
452  {
453  const utils::ustring& sManualLine = *iterManual;
454 
455  DEBUG_LOG(" Line : '" + utils::unicode_to_UTF8(sManualLine) + "'");
456  // Make a temporary line array
457  std::vector<line> lLines;
458  line mLine; mLine.fWidth = 0.0f;
459  std::map<uint, format> lTempFormatList;
460 
461  DEBUG_LOG(" Read chars");
462  for (auto iterChar1 = sManualLine.begin(); iterChar1 != sManualLine.end(); ++iterChar1)
463  {
464  DEBUG_LOG(" char '" + utils::to_string(*iterChar1) + "'");
465  DEBUG_LOG(" Read format");
466  // Read format tags
467  if (*iterChar1 == U'|' && bFormattingEnabled_)
468  {
469  ++iterChar1;
470  if (iterChar1 != sManualLine.end())
471  {
472  if (*iterChar1 != U'|')
473  {
474  get_format(iterChar1, lTempFormatList[uiCounter+mLine.sCaption.size()]);
475  continue;
476  }
477  }
478  else
479  break;
480  }
481 
482  DEBUG_LOG(" Get width");
483  if (*iterChar1 == U' ')
484  mLine.fWidth += fSpaceWidth_ + fTracking_;
485  else if (*iterChar1 == U'\t')
486  mLine.fWidth += 4*fSpaceWidth_ + fTracking_;
487  else
488  {
489  mLine.fWidth += get_character_width(*iterChar1) + fTracking_;
490  auto iterNext = iterChar1 + 1;
491  if (iterNext != sManualLine.end())
492  {
493  if (*iterNext != U' ')
494  mLine.fWidth += get_character_kerning(*iterChar1, *iterNext);
495  }
496  }
497 
498  mLine.sCaption += *iterChar1;
499 
500  if (mLine.fWidth > fBoxW_)
501  {
502  DEBUG_LOG(" Box break " + utils::to_string(mLine.fWidth) + " > " + utils::to_string(fBoxW_));
503  // Whoops, the line is too long...
504  if (mLine.sCaption.find(U" ") != mLine.sCaption.npos && bWordWrap_)
505  {
506  DEBUG_LOG(" Spaced");
507  // There are several words on this line, we'll
508  // be able to put the last one on the next line
509  utils::ustring::iterator iterChar2 = mLine.sCaption.end();
510  utils::ustring sErasedString;
511  uint uiCharToErase = 0;
512  float fErasedWidth = 0.0f;
513  bool bLastWasWord = false;
514  while (mLine.fWidth > fBoxW_ && iterChar2 != mLine.sCaption.begin())
515  {
516  --iterChar2;
517  if (*iterChar2 == U' ')
518  {
519  if (!bLastWasWord || bRemoveStartingSpaces_ || mLine.fWidth - fErasedWidth > fBoxW_)
520  {
521  mLine.fWidth -= fErasedWidth + fSpaceWidth_ + fTracking_;
522  sErasedString.insert(sErasedString.begin(), *iterChar2);
523  fErasedWidth = 0.0f;
524  ++uiCharToErase;
525  }
526  else
527  break;
528  }
529  else
530  {
531  fErasedWidth += get_character_width(*iterChar2) + fTracking_;
532  sErasedString.insert(sErasedString.begin(), *iterChar2);
533  ++uiCharToErase;
534  bLastWasWord = true;
535  }
536  }
537 
538  if (bRemoveStartingSpaces_)
539  {
540  while (iterChar2 != mLine.sCaption.end() && *iterChar2 == U' ')
541  {
542  --uiCharToErase;
543  sErasedString.erase(0, 1);
544  ++iterChar2;
545  }
546  }
547 
548  mLine.sCaption.erase(mLine.sCaption.size() - uiCharToErase, uiCharToErase);
549 
550  lLines.push_back(mLine);
551  for (auto& mFormat : lTempFormatList)
552  lFormatList_.insert(std::move(mFormat));
553 
554  lTempFormatList.clear();
555  uiCounter += mLine.sCaption.size();
556  mLine.fWidth = get_string_width(sErasedString);
557  mLine.sCaption = sErasedString;
558  }
559  else
560  {
561  DEBUG_LOG(" Single word");
562  // There is only one word on this line, or word
563  // wrap is disabled. Anyway, this line is just
564  // too long for the text box : our only option
565  // is to truncate it.
566  if (bAddEllipsis_)
567  {
568  DEBUG_LOG(" Ellipsis");
569  float fWordWidth = 3.0f*(get_character_width(U'.') + fTracking_);
570  utils::ustring::iterator iterChar2 = mLine.sCaption.end();
571  uint uiCharToErase = 0;
572  while ((mLine.fWidth + fWordWidth > fBoxW_) && (iterChar2 != mLine.sCaption.begin()))
573  {
574  --iterChar2;
575  mLine.fWidth -= get_character_width(*iterChar2) + fTracking_;
576  ++uiCharToErase;
577  }
578 
579  DEBUG_LOG(" Char to erase : " + utils::to_string(uiCharToErase) + " / "
580  + utils::to_string(mLine.sCaption.size()));
581 
582  mLine.sCaption.erase(mLine.sCaption.size() - uiCharToErase, uiCharToErase);
583  mLine.sCaption += U"...";
584  }
585  else
586  {
587  DEBUG_LOG(" Truncate");
588  utils::ustring::iterator iterChar2 = mLine.sCaption.end();
589  uint uiCharToErase = 0;
590  while (mLine.fWidth > fBoxW_ && iterChar2 != mLine.sCaption.begin())
591  {
592  --iterChar2;
593  mLine.fWidth -= get_character_width(*iterChar2) + fTracking_;
594  ++uiCharToErase;
595  }
596  mLine.sCaption.erase(mLine.sCaption.size() - uiCharToErase, uiCharToErase);
597  }
598 
599  if (!bWordWrap_)
600  {
601  DEBUG_LOG(" Display single line");
602  // Word wrap is disabled, so we can only display one line anyway.
603  lLineList_.push_back(mLine);
604  for (const auto& mFormat : lTempFormatList)
605  lFormatList_[mFormat.first] = mFormat.second;
606 
607  return;
608  }
609 
610  DEBUG_LOG(" Continue");
611  auto iterTemp = iterChar1;
612  size_t pos = sManualLine.find(U" ", iterChar1 - sManualLine.begin());
613  if (pos == sManualLine.npos)
614  iterChar1 = sManualLine.end();
615  else
616  iterChar1 = sManualLine.begin() + pos;
617 
618  if (iterChar1 != sManualLine.end())
619  {
620  // Read cutted format tags
621  if (bFormattingEnabled_)
622  {
623  while (iterTemp != iterChar1)
624  {
625  if (*iterTemp == U'|')
626  {
627  ++iterTemp;
628  if (iterTemp != iterChar1 && *iterTemp != U'|')
629  get_format(iterTemp, lTempFormatList[uiCounter+mLine.sCaption.size()]);
630  }
631  ++iterTemp;
632  }
633  }
634 
635  // Look for the next word
636  while (iterChar1 != sManualLine.end())
637  {
638  if ((*iterChar1) == U' ')
639  ++iterChar1;
640  else
641  break;
642  }
643 
644  // Add the line
645  if (iterChar1 != sManualLine.end())
646  {
647  --iterChar1;
648  lLines.push_back(mLine);
649  uiCounter += mLine.sCaption.size();
650 
651  for (auto& mFormat : lTempFormatList)
652  lFormatList_.insert(std::move(mFormat));
653 
654  lTempFormatList.clear();
655  mLine.fWidth = 0.0f;
656  mLine.sCaption.clear();
657  }
658  else
659  break;
660  }
661  else
662  break;
663  }
664  }
665  }
666 
667  DEBUG_LOG(" End");
668 
669  if (iterManual != lManualLineList.end() - 1)
670  mLine.sCaption += U"\n";
671 
672  lLines.push_back(mLine);
673  for (auto& mFormat : lTempFormatList)
674  lFormatList_.insert(std::move(mFormat));
675 
676  lTempFormatList.clear();
677  uiCounter += mLine.sCaption.size() - 1;
678 
679  // Add the maximum number of line to this text
680  for (auto& sLine : lLines)
681  {
682  lLineList_.push_back(std::move(sLine));
683  if (lLineList_.size() == uiMaxLineNbr)
684  return;
685  }
686  DEBUG_LOG(" .");
687  }
688  }

773  break;
774  case alignment::RIGHT :
775  fX = fX0 - mLine.fWidth;
776  break;
777  }

789  break;
790  case color_action::RESET :

792  break;
793  default : break;
794  }
795  }
796

798  mLetter.mQuad = pFont_->get_character_bounds(*iterChar) + vector2f(round(fX), round(fY));
799  mLetter.mUVs = pFont_->get_character_uvs(*iterChar);

801  mLetter.bNoRender = (*iterChar == '\n' || *iterChar == ' ' || *iterChar == '\t');

804 
805  if (*iterChar == '\n') continue; // Don't increase the uiCounter
806 
807  float fKerning = 0.0f;

811  fKerning = get_character_kerning(*iterChar, *iterNext);
812 
813  fX += pFont_->get_character_width(*iterChar) + fKerning + fTracking_;
814  ++uiCounter;
815  }
816 
817  fY += get_line_height()*fLineSpacing_;
818  }
819  }
820  else
821  {
822  fW_ = 0.0f;
823  fH_ = 0.0f;
824  }
