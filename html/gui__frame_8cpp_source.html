998 void frame::define_script(const std::string& sScriptName, const std::string& sContent, const std::string& sFile, uint uiLineNbr)
999 {
1000  std::string sCutScriptName = sScriptName;
1001  sCutScriptName.erase(0, 2);
1002 
1003  std::map<std::string, handler>::iterator iterH = lDefinedHandlerList_.find(sCutScriptName);
1004  if (iterH != lDefinedHandlerList_.end())
1005  lDefinedHandlerList_.erase(iterH);
1006 
1007  std::string sAdjustedName = sScriptName;
1008  for (auto iter = sAdjustedName.begin(); iter != sAdjustedName.end(); ++iter)
1009  {
1010  if ('A' <= *iter && *iter <= 'Z')
1011  {
1012  *iter = tolower(*iter);
1013  if (iter != sAdjustedName.begin())
1014  iter = sAdjustedName.insert(iter, '_');
1015  }
1016  }
1017 
1018  std::string sStr;
1019  sStr += "function " + sLuaName_ + ":" + sAdjustedName + "() " + sContent + " end";
1020 
1021  // Use XML specific error handling
1022  lua::state* pLua = pManager_->get_lua();
1023 
1024  std::string sOldFile = pLua->get_global_string("_xml_file_name", false, "");
1025  uint uiOldLineNbr = pLua->get_global_int("_xml_line_nbr", false, 0);
1026  lua::c_function pErrorFunc = pLua->get_lua_error_function();
1027 
1028  pLua->push_string(sFile); pLua->set_global("_xml_file_name");
1029  pLua->push_number(uiLineNbr); pLua->set_global("_xml_line_nbr");
1030  pLua->set_lua_error_function(l_xml_error);
1031 
1032  // Actually register the function
1033  try
1034  {
1035  pLua->do_string(sStr);
1036  lDefinedScriptList_[sCutScriptName] = sContent;
1037  lXMLScriptInfoList_[sCutScriptName].sFile = sFile;
1038  lXMLScriptInfoList_[sCutScriptName].uiLineNbr = uiLineNbr;
1039  }
1040  catch (const lua::exception& e)
1041  {
1042  std::string sError = e.get_description();
1043 
1044  // There is no way (at least in lua 5.1) to use a custom error
1045  // function for syntax error checking (lua_load)...
1046  // Here we hack the error string to recover the necessary informations
1047  // and output a better error message.
1048  if (sError[0] == '[')
1049  {
1050  size_t pos = sError.find("]");
1051  sError.erase(0, pos+2);
1052 
1053  pos = sError.find(":");
1054  uint uiLuaLineNbr = utils::string_to_uint(sError.substr(0, pos));
1055 
1056  sError.erase(0, pos+1);
1057 
1058  sError = sFile + ":" + utils::to_string(uiLineNbr + uiLuaLineNbr - 1) + ":" + sError;
1059  }
1060 
1061  gui::out << gui::error << sError << std::endl;
1062 
1063  event mEvent("LUA_ERROR");
1064  mEvent.add(sError);
1065  pManager_->get_event_manager()->fire_event(mEvent);
1066  }
1067 
1068  pLua->push_string(sOldFile); pLua->set_global("_xml_file_name");
1069  pLua->push_number(uiOldLineNbr); pLua->set_global("_xml_line_nbr");
1070  pLua->set_lua_error_function(pErrorFunc);
1071 }
