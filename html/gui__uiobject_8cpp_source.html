793 void uiobject::read_anchors_(float& iLeft, float& iRight, float& iTop,
794  float& iBottom, float& iXCenter, float& iYCenter) const
795 {
796  iLeft = +std::numeric_limits<float>::infinity();
797  iRight = -std::numeric_limits<float>::infinity();
798  iTop = +std::numeric_limits<float>::infinity();
799  iBottom = -std::numeric_limits<float>::infinity();
800 
801  for (const auto& mOptAnchor : lAnchorList_)
802  {
803  if (!mOptAnchor)
804  continue;
805 
806  const anchor& mAnchor = mOptAnchor.value();
807 
808  // Make sure the anchored object has its borders updated
809  const uiobject* pObj = mAnchor.get_parent();
810  if (pObj)
811  pObj->update_borders_();
812 
813  switch (mAnchor.get_point())
814  {
815  case anchor_point::TOPLEFT :
816  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
817  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
818  break;
819  case anchor_point::TOP :
820  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
821  iXCenter = mAnchor.get_abs_x();
822  break;
823  case anchor_point::TOPRIGHT :
824  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
825  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
826  break;
827  case anchor_point::RIGHT :
828  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
829  iYCenter = mAnchor.get_abs_y();
830  break;
831  case anchor_point::BOTTOMRIGHT :
832  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
833  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
834  break;
835  case anchor_point::BOTTOM :
836  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
837  iXCenter = mAnchor.get_abs_x();
838  break;
839  case anchor_point::BOTTOMLEFT :
840  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
841  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
842  break;
843  case anchor_point::LEFT :
844  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
845  iYCenter = mAnchor.get_abs_y();
846  break;
847  case anchor_point::CENTER :
848  iXCenter = mAnchor.get_abs_x();
849  iYCenter = mAnchor.get_abs_y();
850  break;
851  }

921 void uiobject::update_anchors()
922 {
923  if (!bUpdateAnchors_)
924  return;
925 
926  std::vector<const uiobject*> lAnchorParentList;
927  for (auto& mAnchor : lAnchorList_)
928  {
929  if (!mAnchor)
930  continue;
931 
932  const uiobject* pObj = mAnchor->get_parent();
933  if (pObj)
934  {
935  if (pObj->depends_on(this))
936  {
937  gui::out << gui::error << "gui::" << lType_.back() << " : Cyclic anchor dependency !"
938  << "\"" << sName_ << "\" and \"" << pObj->get_name() << "\" depend on"
939  "eachothers (directly or indirectly).\n\""
940  << anchor::get_string_point(mAnchor->get_point()) << "\" anchor removed." << std::endl;
941 
942  mAnchor.reset();
943  continue;
944  }
945 
946  if (utils::find(lAnchorParentList, pObj) == lAnchorParentList.end())
947  lAnchorParentList.push_back(pObj);
948  }
949  }
950 
951  for (const auto* pParent : lPreviousAnchorParentList_)
952  {
953  if (utils::find(lAnchorParentList, pParent) == lAnchorParentList.end())
954  pParent->notify_anchored_object(this, false);
955  }
956 
957  for (const auto* pParent : lAnchorParentList)
958  {
959  if (utils::find(lPreviousAnchorParentList_, pParent) == lPreviousAnchorParentList_.end())
960  pParent->notify_anchored_object(this, true);
961  }
962 
963  lPreviousAnchorParentList_ = std::move(lAnchorParentList);
964  bUpdateAnchors_ = false;
965 }

1076 void uiobject::clear_links_()
1077 {
1078  if (!bVirtual_)
1079  {
1080  // Tell this widget's anchor parents that it is no longer anchored to them
1081  for (auto& mAnchor : lAnchorList_)
1082  {
1083  if (mAnchor && mAnchor->get_parent())
1084  mAnchor->get_parent()->notify_anchored_object(this, false);
1085 
1086  mAnchor.reset();
1087  }
1088 
1089  // Replace anchors pointing to this widget by absolute anchors
1090  // (need to copy the anchored object list, because the objects will attempt to modify it when
1091  // un-anchored, which would invalidate our iteration)
1092  std::vector<uiobject*> lTempAnchoredObjectList = std::move(lAnchoredObjectList_);
1093  for (auto* pObj : lTempAnchoredObjectList)
1094  {
1095  std::vector<anchor_point> lAnchoredPointList;
1096  for (const auto& mAnchor : pObj->get_point_list())
1097  {
1098  if (mAnchor && mAnchor->get_parent() == this)
1099  lAnchoredPointList.push_back(mAnchor->get_point());
1100  }
1101 
1102  for (const auto& mPoint : lAnchoredPointList)
1103  {
1104  const anchor* pAnchor = pObj->get_point(mPoint);
1105  anchor mNewAnchor = anchor(pObj, mPoint, "", anchor_point::TOPLEFT);
1106  vector2i mOffset = pAnchor->get_abs_offset();
1107 
1108  switch (pAnchor->get_parent_point())
1109  {
1110  case anchor_point::TOPLEFT : mOffset += lBorderList_.top_left(); break;
1111  case anchor_point::TOP : mOffset.y += lBorderList_.top; break;
1112  case anchor_point::TOPRIGHT : mOffset += lBorderList_.top_right(); break;
1113  case anchor_point::RIGHT : mOffset.x += lBorderList_.right; break;
1114  case anchor_point::BOTTOMRIGHT : mOffset += lBorderList_.bottom_right(); break;
1115  case anchor_point::BOTTOM : mOffset.y += lBorderList_.bottom; break;
1116  case anchor_point::BOTTOMLEFT : mOffset += lBorderList_.bottom_left(); break;
1117  case anchor_point::LEFT : mOffset.x += lBorderList_.left; break;
1118  case anchor_point::CENTER : mOffset += lBorderList_.center(); break;
1119  }
1120 
1121  mNewAnchor.set_abs_offset(mOffset);
1122  pObj->set_point(mNewAnchor);
1123  }
1124 
1125  pObj->update_anchors();
1126  }
1127  }
1128 
1129  // Unregister this object from the GUI manager
1130  pManager_->remove_uiobject(this);
1131
