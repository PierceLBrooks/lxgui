846 void uiobject::make_borders_(float& fMin, float& fMax, float fCenter, float fSize) const
847 {
848  if (std::isinf(fMin) && std::isinf(fMax))
849  {
850  if (!std::isinf(fSize) && fSize > 0.0f && !std::isinf(fCenter))
851  {
852  fMin = fCenter - fSize/2.0f;
853  fMax = fCenter + fSize/2.0f;
854  }
855  else
856  bReady_ = false;
857  }
858  else if (std::isinf(fMax))
859  {
860  if (!std::isinf(fSize) && fSize > 0.0f)
861  fMax = fMin + fSize;
862  else if (!std::isinf(fCenter))
863  fMax = fMin + 2.0f*(fCenter - fMin);
864  else
865  bReady_ = false;
866  }
867  else if (std::isinf(fMin))
868  {
869  if (!std::isinf(fSize) && fSize > 0.0f)
870  fMin = fMax - fSize;
871  else if (!std::isinf(fCenter))
872  fMin = fMax - 2.0f*(fMax - fCenter);
873  else
874  bReady_ = false;
875  }

878 void uiobject::read_anchors_(float& fLeft, float& fRight, float& fTop,
879  float& fBottom, float& fXCenter, float& fYCenter) const
880 {
881  fLeft = +std::numeric_limits<float>::infinity();
882  fRight = -std::numeric_limits<float>::infinity();
883  fTop = +std::numeric_limits<float>::infinity();
884  fBottom = -std::numeric_limits<float>::infinity();
885 
886  for (const auto& mOptAnchor : lAnchorList_)
887  {
888  if (!mOptAnchor)
889  continue;
890 
891  const anchor& mAnchor = mOptAnchor.value();
892 
893  // Make sure the anchored object has its borders updated
894  const uiobject* pObj = mAnchor.get_parent();
895  if (pObj)
896  pObj->update_borders_();
897 
898  switch (mAnchor.get_point())
899  {
900  case anchor_point::TOPLEFT :
901  fTop = std::min<float>(fTop, mAnchor.get_abs_y());
902  fLeft = std::min<float>(fLeft, mAnchor.get_abs_x());
903  break;
904  case anchor_point::TOP :
905  fTop = std::min<float>(fTop, mAnchor.get_abs_y());
906  fXCenter = mAnchor.get_abs_x();
907  break;
908  case anchor_point::TOPRIGHT :
909  fTop = std::min<float>(fTop, mAnchor.get_abs_y());
910  fRight = std::max<float>(fRight, mAnchor.get_abs_x());
911  break;
912  case anchor_point::RIGHT :
913  fRight = std::max<float>(fRight, mAnchor.get_abs_x());
914  fYCenter = mAnchor.get_abs_y();
915  break;
916  case anchor_point::BOTTOMRIGHT :
917  fBottom = std::max<float>(fBottom, mAnchor.get_abs_y());
918  fRight = std::max<float>(fRight, mAnchor.get_abs_x());
919  break;
920  case anchor_point::BOTTOM :
921  fBottom = std::max<float>(fBottom, mAnchor.get_abs_y());
922  fXCenter = mAnchor.get_abs_x();
923  break;
924  case anchor_point::BOTTOMLEFT :
925  fBottom = std::max<float>(fBottom, mAnchor.get_abs_y());
926  fLeft = std::min<float>(fLeft, mAnchor.get_abs_x());
927  break;
928  case anchor_point::LEFT :
929  fLeft = std::min<float>(fLeft, mAnchor.get_abs_x());
930  fYCenter = mAnchor.get_abs_y();
931  break;
932  case anchor_point::CENTER :
933  fXCenter = mAnchor.get_abs_x();
934  fYCenter = mAnchor.get_abs_y();
935  break;
936  }
937  }
938 }

940 void uiobject::update_borders_() const
941 {
942  if (!bUpdateBorders_)
943  return;
944 
945  //#define DEBUG_LOG(msg) gui::out << (msg) << std::endl
946  #define DEBUG_LOG(msg)
947 
948  bool bOldReady = bReady_;
949  bReady_ = true;
950 
951  if (!lAnchorList_.empty())
952  {
953  float fLeft = 0.0f, fRight = 0.0f, fTop = 0.0f, fBottom = 0.0f;
954  float fXCenter = 0.0f, fYCenter = 0.0f;
955 
956  float fRoundedWidth = round_to_pixel(fAbsWidth_, utils::rounding_method::NEAREST_NOT_ZERO);
957  float fRoundedHeight = round_to_pixel(fAbsHeight_, utils::rounding_method::NEAREST_NOT_ZERO);
958 
959  DEBUG_LOG(" Read anchors");
960  read_anchors_(fLeft, fRight, fTop, fBottom, fXCenter, fYCenter);
961 
962  DEBUG_LOG(" Make borders");
963  make_borders_(fTop, fBottom, fYCenter, fRoundedHeight);
964  make_borders_(fLeft, fRight, fXCenter, fRoundedWidth);
965 
966  if (bReady_)
967  {
968  if (fRight < fLeft)
969  fRight = fLeft+1;
970  if (fBottom < fTop)
971  fBottom = fTop+1;
972 
973  lBorderList_ = bounds2f(fLeft, fRight, fTop, fBottom);
974  }
975  else
976  lBorderList_ = bounds2f::ZERO;
977 
978  bUpdateBorders_ = false;
979  }
980  else
981  {
982  lBorderList_ = bounds2f(0.0, 0.0, fAbsWidth_, fAbsHeight_);
983  bReady_ = false;
984  }
985 
986  lBorderList_.left = round_to_pixel(lBorderList_.left);
987  lBorderList_.right = round_to_pixel(lBorderList_.right);
988  lBorderList_.top = round_to_pixel(lBorderList_.top);
989  lBorderList_.bottom = round_to_pixel(lBorderList_.bottom);
990 
991  if (bReady_ || (!bReady_ && bOldReady))
992  {
993  DEBUG_LOG(" Fire redraw");
994  notify_renderer_need_redraw();
995  }
996  DEBUG_LOG(" @");
997 }

999 void uiobject::update_anchors_()
1000 {
1001  std::vector<const uiobject*> lAnchorParentList;
1002  for (auto& mAnchor : lAnchorList_)
1003  {
1004  if (!mAnchor)
1005  continue;
1006 
1007  const uiobject* pObj = mAnchor->get_parent();
1008  if (pObj)
1009  {
1010  if (pObj->depends_on(this))
1011  {
1012  gui::out << gui::error << "gui::" << lType_.back() << " : Cyclic anchor dependency ! "
1013  << "\"" << sName_ << "\" and \"" << pObj->get_name() << "\" depend on "
1014  "eachothers (directly or indirectly).\n\""
1015  << anchor::get_string_point(mAnchor->get_point()) << "\" anchor removed." << std::endl;
1016 
1017  mAnchor.reset();
1018  continue;
1019  }
1020 
1021  if (utils::find(lAnchorParentList, pObj) == lAnchorParentList.end())
1022  lAnchorParentList.push_back(pObj);
1023  }
1024  }
1025 
1026  for (const auto* pParent : lPreviousAnchorParentList_)
1027  {
1028  if (utils::find(lAnchorParentList, pParent) == lAnchorParentList.end())
1029  pParent->notify_anchored_object(this, false);
1030  }
1031 
1032  for (const auto* pParent : lAnchorParentList)
1033  {
1034  if (utils::find(lPreviousAnchorParentList_, pParent) == lPreviousAnchorParentList_.end())
1035  pParent->notify_anchored_object(this, true);
1036  }
1037 
1038  lPreviousAnchorParentList_ = std::move(lAnchorParentList);
1039 }
