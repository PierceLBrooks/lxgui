808 void uiobject::read_anchors_(float& iLeft, float& iRight, float& iTop,
809  float& iBottom, float& iXCenter, float& iYCenter) const
810 {
811  iLeft = +std::numeric_limits<float>::infinity();
812  iRight = -std::numeric_limits<float>::infinity();
813  iTop = +std::numeric_limits<float>::infinity();
814  iBottom = -std::numeric_limits<float>::infinity();

824  {

826  iTop = std::min<float>(iTop, mAnchor.get_abs_y());
827  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
828  break;
829  case anchor_point::TOP :
830  iTop = std::min<float>(iTop, mAnchor.get_abs_y());

832  break;
833  case anchor_point::TOPRIGHT :
834  iTop = std::min<float>(iTop, mAnchor.get_abs_y());

836  break;
837  case anchor_point::RIGHT :
838  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
839  iYCenter = mAnchor.get_abs_y();
840  break;
841  case anchor_point::BOTTOMRIGHT :
842  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
843  iRight = std::max<float>(iRight, mAnchor.get_abs_x());
844  break;
845  case anchor_point::BOTTOM :
846  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
847  iXCenter = mAnchor.get_abs_x();
848  break;
849  case anchor_point::BOTTOMLEFT :
850  iBottom = std::max<float>(iBottom, mAnchor.get_abs_y());
851  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
852  break;
853  case anchor_point::LEFT :
854  iLeft = std::min<float>(iLeft, mAnchor.get_abs_x());
855  iYCenter = mAnchor.get_abs_y();
856  break;
857  case anchor_point::CENTER :

931 void uiobject::update_anchors()
932 {

934  {
935  std::vector<std::map<anchor_point, anchor>::iterator> lEraseList;
936  for (auto iterAnchor : utils::range::iterator(lAnchorList_))
937  {
938  const uiobject* pObj = iterAnchor->second.get_parent();
939  if (pObj && pObj->depends_on(this))

945

948  }
949

951  lAnchorList_.erase(iterErase);

954  for (const auto& mAnchor : utils::range::value(lAnchorList_))
955  {
956  const uiobject* pParent = mAnchor.get_parent();
957  if (pParent && utils::find(lAnchorParentList, pParent) == lAnchorParentList.end())
958  lAnchorParentList.push_back(pParent);

962  {
963  if (utils::find(lAnchorParentList, pParent) == lAnchorParentList.end())
964  pParent->notify_anchored_object(this, false);

966 
967  for (const auto* pParent : lAnchorParentList)
968  {
969  if (utils::find(lPreviousAnchorParentList_, pParent) == lPreviousAnchorParentList_.end())
970  pParent->notify_anchored_object(this, true);
