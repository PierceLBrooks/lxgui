29 void frame::parse_attributes_(xml::block* pBlock)
30 {
31  if (pBlock->is_provided("hidden") || !bInherits_)
32  set_shown(!utils::string_to_bool(pBlock->get_attribute("hidden")));
33 
34  if ((pBlock->is_provided("setAllPoints") || !bInherits_) &&
35  (utils::string_to_bool(pBlock->get_attribute("setAllPoints"))))
36  set_all_points("$parent");
37 
38  if (pBlock->is_provided("alpha") || !bInherits_)
39  set_alpha(utils::string_to_float(pBlock->get_attribute("alpha")));
40  if (pBlock->is_provided("topLevel") || !bInherits_)
41  set_top_level(utils::string_to_bool(pBlock->get_attribute("topLevel")));
42  if (pBlock->is_provided("movable") || !bInherits_)
43  set_movable(utils::string_to_bool(pBlock->get_attribute("movable")));
44  if (pBlock->is_provided("resizable") || !bInherits_)
45  set_resizable(utils::string_to_bool(pBlock->get_attribute("resizable")));
46  if (pBlock->is_provided("frameStrata") || !bInherits_)
47  set_frame_strata(pBlock->get_attribute("frameStrata"));
48  if (pBlock->is_provided("frameLevel"))
49  {
50  if (!bVirtual_)
51  {
52  std::string sFrameLevel = pBlock->get_attribute("frameLevel");
53  if (sFrameLevel != "PARENT")
54  set_level(utils::string_to_int(sFrameLevel));
55  }
56  else
57  {
58  gui::out << gui::warning << pBlock->get_location() << " : "
59  << "\"frameLevel\" is not allowed for virtual widgets. Ignored." << std::endl;
60  }
61  }
62  if (pBlock->is_provided("enableMouse") || !bInherits_)
63  enable_mouse(utils::string_to_bool(pBlock->get_attribute("enableMouse")));
64  if (pBlock->is_provided("enableMouseWheel") || !bInherits_)
65  enable_mouse_wheel(utils::string_to_bool(pBlock->get_attribute("enableMouseWheel")));
66  if (pBlock->is_provided("enableKeyboard") || !bInherits_)
67  enable_keyboard(utils::string_to_bool(pBlock->get_attribute("enableKeyboard")));
68  if (pBlock->is_provided("clampedToScreen") || !bInherits_)
69  set_clamped_to_screen(utils::string_to_bool(pBlock->get_attribute("clampedToScreen")));
70 }

253  if (pLayersBlock)
254  {
255  for (auto* pLayerBlock : pLayersBlock->blocks())
256  {
257  std::string sLevel = pLayerBlock->get_attribute("level");
258  for (auto* pRegionBlock : pLayerBlock->blocks())
259  {
260  std::unique_ptr<layered_region> pRegion = pManager_->create_layered_region(pRegionBlock->get_name());
261  if (!pRegion)
262  continue;
263 
264  try
265  {
266  pRegion->set_parent(this);
267  pRegion->set_draw_layer(sLevel);
268  pRegion->parse_block(pRegionBlock);
269  add_region(std::move(pRegion));
270  }
271  catch (const exception& e)
272  {
273  gui::out << gui::error << e.get_description() << std::endl;
274  }
275  catch (...)
276  {
277  throw;
278  }
279  }
