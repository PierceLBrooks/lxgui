103 int lua_check_button::_set_checked(lua_State* pLua)
104 {
105  if (!check_object_())
106  return 0;
107 
108  // This function is a bit trickier, as it doesn't require any particular
109  // type for its argument. Whatever it is, Lua can convert it to a boolean
110  // value, and we use this value to check/uncheck the button.
111 
112  if (lua_gettop(pLua) > 0)
113  {
114  if (lua_toboolean(pLua, 1) == 0)
115  get_object()->uncheck();
116  else
117  get_object()->check();
118  }
119  else
120  get_object()->check();
121 
122  return 0;
123 }
