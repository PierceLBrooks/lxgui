110 void slider::constrain_thumb_()
111 {
112  if (fMaxValue_ == fMinValue_)
113  return;
114 
115  if ((mOrientation_ == orientation::HORIZONTAL && uiAbsWidth_ != 0) ||
116  (mOrientation_ == orientation::VERTICAL && uiAbsHeight_ != 0))
117  {
118  float fValue = 0.0f;
119 
120  if (bThumbMoved_)
121  {
122  if (mOrientation_ == orientation::HORIZONTAL)
123  fValue = float(pThumbTexture_->get_point(anchor_point::CENTER)->get_abs_offset_x())/float(uiAbsWidth_);
124  else
125  fValue = float(pThumbTexture_->get_point(anchor_point::CENTER)->get_abs_offset_y())/float(uiAbsHeight_);
126 
127  fValue *= (fMaxValue_ - fMinValue_);
128  fValue += fMinValue_;
129  fValue = fValue > fMaxValue_ ? fMaxValue_ : (fValue < fMinValue_ ? fMinValue_ : fValue);
130  step_value(fValue, fValueStep_);
131  }
132 
133  float fCoef = (fValue - fMinValue_)/(fMaxValue_ - fMinValue_);
134 
135  anchor* pAnchor = pThumbTexture_->modify_point(anchor_point::CENTER);
136  if (mOrientation_ == orientation::HORIZONTAL)
137  pAnchor->set_abs_offset(uiAbsWidth_*fCoef, 0);
138  else
139  pAnchor->set_abs_offset(0, uiAbsHeight_*fCoef);
140  }

143 void slider::on_event(const event& mEvent)
144 {
145  alive_checker mChecker(this);
146 
147  frame::on_event(mEvent);
148  if (!mChecker.is_alive())
149  return;
150 
151  if (bIsMouseEnabled_ && pManager_->is_input_enabled())
152  {
153  if (mEvent.get_name() == "MOUSE_PRESSED")
154  {
155  if (bMouseInThumb_)
156  {
157  pManager_->start_moving(
158  pThumbTexture_, pThumbTexture_->modify_point(anchor_point::CENTER),
159  mOrientation_ == orientation::HORIZONTAL ? constraint::X : constraint::Y,
160  std::bind(&slider::constrain_thumb_, this)
161  );
162  bThumbMoved_ = true;
163  }
164  else if (bMouseInFrame_ && bAllowClicksOutsideThumb_)
165  {
166  float fValue;
167  if (mOrientation_ == orientation::HORIZONTAL)
168  {
169  float fOffset = float(iMousePosX_ - lBorderList_.left);
170  fValue = fOffset/uiAbsWidth_;
171  set_value(fValue*(fMaxValue_ - fMinValue_) + fMinValue_);
172  }
173  else
174  {
175  float fOffset = float(iMousePosY_ - lBorderList_.top);
176  fValue = fOffset/uiAbsHeight_;
177  set_value(fValue*(fMaxValue_ - fMinValue_) + fMinValue_);
178  }
179 
180  if (pThumbTexture_)
181  {
182  float fCoef = (fValue_ - fMinValue_)/(fMaxValue_ - fMinValue_);
183 
184  anchor* pAnchor = pThumbTexture_->modify_point(anchor_point::CENTER);
185  if (mOrientation_ == orientation::HORIZONTAL)
186  pAnchor->set_abs_offset(uiAbsWidth_*fCoef, 0);
187  else
188  pAnchor->set_abs_offset(0, uiAbsHeight_*fCoef);
189 
190  pManager_->start_moving(
191  pThumbTexture_, pThumbTexture_->modify_point(anchor_point::CENTER),
192  mOrientation_ == orientation::HORIZONTAL ? constraint::X : constraint::Y
193  );
194  bThumbMoved_ = true;
195  }
196  }
197  }
198  else if (mEvent.get_name() == "MOUSE_RELEASED")
199  {
200  if (pThumbTexture_)
201  {
202  pManager_->stop_moving(pThumbTexture_);
203  bThumbMoved_ = false;
204  }
205  }
